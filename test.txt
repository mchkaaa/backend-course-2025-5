# Тестування GET для неіснуючого файлу (має повернути 404)
curl -X GET http://localhost:3000/200 -v

# Тестування PUT - завантаження картинки
curl -X PUT http://localhost:3000/200 --data-binary "@test-image.jpg" -v

# Тестування GET для існуючого файлу (має повернути 200 та картинку)
curl -X GET http://localhost:3000/200 -o downloaded-image.jpg -v

# Тестування DELETE
curl -X DELETE http://localhost:3000/200 -v

# Тестування недозволеного методу (має повернути 405)
curl -X POST http://localhost:3000/200 -v

# Тестування невірного HTTP коду (має повернути 400)
curl -X GET http://localhost:3000/abc -v
